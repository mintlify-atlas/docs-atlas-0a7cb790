The `IosInstances` resource provides methods to create, list, retrieve, and delete iOS instances.

## Methods

### create()

Create a new iOS instance.

```typescript
create(
  params: IosInstanceCreateParams,
  options?: RequestOptions
): APIPromise<IosInstance>
```

#### Parameters

<ParamField query="reuseIfExists" type="boolean" optional>
  If there is another instance with given labels and region, return that one instead of creating a new instance.
</ParamField>

<ParamField query="wait" type="boolean" optional>
  Return after the instance is ready to connect.
</ParamField>

<ParamField body="metadata" type="object" optional>
  Metadata for the iOS instance.
  
  <ParamField body="displayName" type="string" optional>
    Human-readable display name for the instance.
  </ParamField>
  
  <ParamField body="labels" type="Record<string, string>" optional>
    Key-value pairs for labeling the instance.
  </ParamField>
</ParamField>

<ParamField body="spec" type="object" optional>
  Specification for the iOS instance.
  
  <ParamField body="region" type="string" optional>
    The region where the instance will be created. If not given, will be decided based on scheduling clues and availability.
  </ParamField>
  
  <ParamField body="inactivityTimeout" type="string" optional>
    After how many minutes of inactivity should the instance be terminated. Example values: `"1m"`, `"10m"`, `"3h"`. Default is `"3m"`. Providing `"0"` disables inactivity checks altogether.
  </ParamField>
  
  <ParamField body="hardTimeout" type="string" optional>
    After how many minutes should the instance be terminated. Example values: `"1m"`, `"10m"`, `"3h"`. Default is `"0"` which means no hard timeout.
  </ParamField>
  
  <ParamField body="clues" type="Array<Clue>" optional>
    Scheduling clues to help determine optimal instance placement.
    
    Each clue object has:
    - `kind`: `"ClientIP"`
    - `clientIp`: IP address
  </ParamField>
  
  <ParamField body="initialAssets" type="Array<InitialAsset>" optional>
    Assets to load when the instance starts.
    
    Each asset object has:
    - `kind`: `"App"`
    - `source`: `"URL"`, `"AssetName"`, or `"AssetID"`
    - `url`: URL to the app (when source is `"URL"`)
    - `assetName`: Asset name (when source is `"AssetName"`)
    - `assetId`: Asset ID (when source is `"AssetID"`)
    - `launchMode`: `"ForegroundIfRunning"` or `"RelaunchIfRunning"` - specifies how to launch the app after installation. If not given, the app won't be launched.
  </ParamField>
  
  <ParamField body="sandbox" type="object" optional>
    Sandbox configuration.
    
    <ParamField body="xcode" type="object" optional>
      Xcode sandbox configuration.
      
      <ParamField body="enabled" type="boolean" optional>
        Enable Xcode sandbox.
      </ParamField>
    </ParamField>
  </ParamField>
</ParamField>

#### Returns

<ResponseField name="metadata" type="object" required>
  <ResponseField name="id" type="string" required>
    Unique identifier for the instance.
  </ResponseField>
  
  <ResponseField name="createdAt" type="string" required>
    ISO 8601 timestamp when the instance was created.
  </ResponseField>
  
  <ResponseField name="organizationId" type="string" required>
    Organization ID that owns this instance.
  </ResponseField>
  
  <ResponseField name="displayName" type="string" optional>
    Human-readable display name.
  </ResponseField>
  
  <ResponseField name="labels" type="Record<string, string>" optional>
    Key-value labels attached to the instance.
  </ResponseField>
  
  <ResponseField name="terminatedAt" type="string" optional>
    ISO 8601 timestamp when the instance was terminated.
  </ResponseField>
</ResponseField>

<ResponseField name="spec" type="object" required>
  <ResponseField name="inactivityTimeout" type="string" required>
    Configured inactivity timeout.
  </ResponseField>
  
  <ResponseField name="region" type="string" required>
    Region where the instance is running.
  </ResponseField>
  
  <ResponseField name="hardTimeout" type="string" optional>
    Configured hard timeout.
  </ResponseField>
</ResponseField>

<ResponseField name="status" type="object" required>
  <ResponseField name="token" type="string" required>
    Authentication token for connecting to the instance.
  </ResponseField>
  
  <ResponseField name="state" type="string" required>
    Current state: `"unknown"`, `"creating"`, `"assigned"`, `"ready"`, or `"terminated"`.
  </ResponseField>
  
  <ResponseField name="apiUrl" type="string" optional>
    API URL for the instance.
  </ResponseField>
  
  <ResponseField name="mcpUrl" type="string" optional>
    MCP (Model Context Protocol) URL for the instance.
  </ResponseField>
  
  <ResponseField name="endpointWebSocketUrl" type="string" optional>
    WebSocket URL for endpoint connection.
  </ResponseField>
  
  <ResponseField name="targetHttpPortUrlPrefix" type="string" optional>
    URL prefix for accessing HTTP ports on the instance.
  </ResponseField>
  
  <ResponseField name="errorMessage" type="string" optional>
    Error message if the instance failed.
  </ResponseField>
  
  <ResponseField name="sandbox" type="object" optional>
    <ResponseField name="xcode" type="object" optional>
      <ResponseField name="url" type="string" optional>
        Xcode connection URL.
      </ResponseField>
    </ResponseField>
  </ResponseField>
</ResponseField>

#### Example

```typescript
const instance = await limrun.iosInstances.create({
  wait: true,
  metadata: {
    displayName: "iPhone 15 Pro",
    labels: {
      environment: "production",
      team: "mobile"
    }
  },
  spec: {
    region: "us-east",
    inactivityTimeout: "15m",
    hardTimeout: "2h",
    clues: [
      {
        kind: "ClientIP",
        clientIp: "203.0.113.42"
      }
    ],
    initialAssets: [
      {
        kind: "App",
        source: "URL",
        url: "https://example.com/app.ipa",
        launchMode: "ForegroundIfRunning"
      }
    ],
    sandbox: {
      xcode: {
        enabled: true
      }
    }
  }
});

console.log(instance.metadata.id);
console.log(instance.status.state); // "ready"
console.log(instance.status.mcpUrl);
```

### list()

List iOS instances with optional filters.

```typescript
list(
  query?: IosInstanceListParams,
  options?: RequestOptions
): PagePromise<IosInstancesItems, IosInstance>
```

#### Parameters

<ParamField query="labelSelector" type="string" optional>
  Labels filter to apply to instances to return. Expects a comma-separated list of key=value pairs (e.g., `"env=prod,region=us-west"`).
</ParamField>

<ParamField query="region" type="string" optional>
  Region where the instance is scheduled on.
</ParamField>

<ParamField query="state" type="string" optional>
  State filter to apply to instances to return. Each comma-separated state will be used as part of an OR clause, e.g. `"assigned,ready"` will return all instances that are either assigned or ready.
  
  Valid states: `"creating"`, `"assigned"`, `"ready"`, `"terminated"`, `"unknown"`
</ParamField>

<ParamField query="pageSize" type="number" optional>
  Number of items per page.
</ParamField>

<ParamField query="pageToken" type="string" optional>
  Token for pagination.
</ParamField>

#### Returns

Returns a paginated list of `IosInstance` objects.

#### Example

```typescript
// List all ready instances
const instances = await limrun.iosInstances.list({
  state: "ready"
});

for await (const instance of instances) {
  console.log(instance.metadata.id, instance.status.state);
}

// List instances with specific labels in a region
const prodInstances = await limrun.iosInstances.list({
  labelSelector: "environment=production",
  region: "us-east"
});
```

### get()

Get an iOS instance by ID.

```typescript
get(
  id: string,
  options?: RequestOptions
): APIPromise<IosInstance>
```

#### Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the iOS instance.
</ParamField>

#### Returns

Returns an `IosInstance` object with the same structure as described in the `create()` method.

#### Example

```typescript
const instance = await limrun.iosInstances.get("instance_456");

console.log(instance.metadata.displayName);
console.log(instance.status.state);
console.log(instance.status.apiUrl);
```

### delete()

Delete an iOS instance.

```typescript
delete(
  id: string,
  options?: RequestOptions
): APIPromise<void>
```

#### Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the iOS instance to delete.
</ParamField>

#### Returns

Returns a promise that resolves when the instance is deleted.

#### Example

```typescript
await limrun.iosInstances.delete("instance_456");
console.log("Instance deleted successfully");
```

## Type Definitions

### IosInstance

The complete iOS instance object returned by the API.

```typescript
interface IosInstance {
  metadata: {
    id: string;
    createdAt: string;
    organizationId: string;
    displayName?: string;
    labels?: Record<string, string>;
    terminatedAt?: string;
  };
  spec: {
    inactivityTimeout: string;
    region: string;
    hardTimeout?: string;
  };
  status: {
    token: string;
    state: 'unknown' | 'creating' | 'assigned' | 'ready' | 'terminated';
    apiUrl?: string;
    mcpUrl?: string;
    endpointWebSocketUrl?: string;
    errorMessage?: string;
    targetHttpPortUrlPrefix?: string;
    sandbox?: {
      xcode?: {
        url?: string;
      };
    };
  };
}
```

### IosInstanceCreateParams

Parameters for creating a new iOS instance.

```typescript
interface IosInstanceCreateParams {
  // Query parameters
  reuseIfExists?: boolean;
  wait?: boolean;
  
  // Body parameters
  metadata?: {
    displayName?: string;
    labels?: Record<string, string>;
  };
  spec?: {
    region?: string;
    inactivityTimeout?: string;
    hardTimeout?: string;
    clues?: Array<{
      kind: 'ClientIP';
      clientIp?: string;
    }>;
    initialAssets?: Array<{
      kind: 'App';
      source: 'URL' | 'AssetName' | 'AssetID';
      url?: string;
      assetName?: string;
      assetId?: string;
      launchMode?: 'ForegroundIfRunning' | 'RelaunchIfRunning';
    }>;
    sandbox?: {
      xcode?: {
        enabled?: boolean;
      };
    };
  };
}
```
